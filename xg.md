{=== 缠论买卖点选股公式 ===}
{参数设置: 选择买卖点类型 1=买1 2=买2 3=买3 4=卖1 5=卖2 6=卖3}
N_TYPE:1,1,6,1;

{基础数据}
FRAC:=TDXDLL2(2,H,L);
BIZG:=TDXDLL2(5,FRAC,H,L);
BIZD:=TDXDLL2(6,FRAC,H,L);
BISE:=TDXDLL2(7,FRAC,H,L);
BIDIR:=TDXDLL2(8,FRAC,H,L);

{中枢参数}
N:=BARSLAST(BISE=2);
ZG:=IF(N<200,REF(BIZG,N),0);
ZD:=IF(N<200,REF(BIZD,N),0);
DIR:=IF(N<200,REF(BIDIR,N),0);

{买1: 向下中枢结束后笔底低于中枢低}
M1:=FRAC=-1 AND N>0 AND N<200 AND DIR=-1 AND L<ZD;

{卖1: 向上中枢结束后笔顶高于中枢高}
M2:=FRAC=+1 AND N>0 AND N<200 AND DIR=1 AND H>ZG;

{买2: 买1后回调不破买1低}
T1:=BARSLAST(M1);
P1:=IF(T1>0 AND T1<200,REF(L,T1),0);
M3:=FRAC=-1 AND T1>0 AND T1<200 AND L>P1 AND COUNT(M1,T1)=0;

{卖2: 卖1后反弹不破卖1高}
T2:=BARSLAST(M2);
P2:=IF(T2>0 AND T2<200,REF(H,T2),0);
M4:=FRAC=+1 AND T2>0 AND T2<200 AND H<P2 AND COUNT(M2,T2)=0;

{买3: 向上中枢后回调不进入中枢}
M5:=FRAC=-1 AND N>0 AND N<200 AND DIR=1 AND L>ZG;

{卖3: 向下中枢后反弹不进入中枢}
M6:=FRAC=+1 AND N>0 AND N<200 AND DIR=-1 AND H<ZD;

{根据参数输出对应信号}
N_TYPE=1 AND M1 OR N_TYPE=2 AND M3 OR N_TYPE=3 AND M5 OR N_TYPE=4 AND M2 OR N_TYPE=5 AND M4 OR N_TYPE=6 AND M6;
